Program to check if a number is Even or Odd:


    #include 
        using namespace std;
        int main()
        {
            int n;
            cout << "Enter an integer: ";
            cin >> n;
            if ( n % 2 == 0)
            cout << n << " is even.";
            else
            cout << n << " is odd.";
            return 0;
        }
